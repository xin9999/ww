Part4: 模板代码解释

Main.view.xml
基于模板生成的xml视图中 主要包含2个<TextArea> 和<Button> 其中第一个TextArea的内容绑定到controll模型的initialContext属性
第二TextArea内容绑定到apiResponse属性. 而button 的press事件绑定controller中对应的function startWorkflowInstance

<mvc:View controllerName="zccs.uicreditlimitapply.controller.Main"
    xmlns:mvc="sap.ui.core.mvc" displayBlock="true"
    xmlns="sap.m">
    <App id="app">
        <pages>
            <Page title="Workflow Start UI">
                <content>
                    <VBox width="100%" direction="Column" class="sapUiTinyMargin">
                        <items>
                            <Text text="Initial Workflow Context:"/>
                            <TextArea width="100%" value="{/initialContext}" rows="8" />
                            <Button text="Start Workflow" press="startWorkflowInstance"/>
                            <Text text="API Response:" class="sapUiSmallMarginTop"/>
                            <TextArea width="100%" value="{/apiResponse}" rows="14"/>
                        </items>
                    </VBox>
                </content>
            </Page>
        </pages>
    </App>
</mvc:View>

在Controller中初始化了一个 JSON模型,该data model初始虎 2个属性initialContext 和 apiResponse对应到UI的TextArea.
当ui点击Start Workflow按钮, 触发对应的function startWorkflowInstance. 该function通过ajax post对workflow进行API请求
请求首先通过ajax发送get请求this._fetchToken()获取x-csrf token, 再发送post请求到url: this._getWorkflowRuntimeBaseURL() + "/workflow-instances", 该请求的body带有数据data 来自与页面TextArea通过 view-model的 two-way 绑定传递数据. async: false 表明发送的是同步ajax, 请求成功/失败,都会将workflow响应的内容存储到 初始化定义的defaul model的/apiResponse中, 再传递到同样绑定该属性ui的TextArea中.
在start ui 和 workflow通讯中,可以自行约定传递的json格式内容.

Main.controll.js

sap.ui.define(
  ["sap/ui/core/mvc/Controller"],
  /**
   * @param {typeof sap.ui.core.mvc.Controller} Controller
   */
  function (Controller) {
    "use strict";

    return Controller.extend(
      "zccs.uicreditlimitapply.controller.Main",
      {
        onInit: function () {
          this.getView().setModel(
            new sap.ui.model.json.JSONModel({
              initialContext: JSON.stringify(
                { someProperty: "some value" },
                null,
                4
              ),
              apiResponse: "",
            })
          );
        },

        startWorkflowInstance: function () {
          var model = this.getView().getModel();
          var definitionId = "z.wfwf";
          var initialContext = model.getProperty("/initialContext");

          var data = {
            definitionId: definitionId,
            context: JSON.parse(initialContext),
          };

          $.ajax({
            url: this._getWorkflowRuntimeBaseURL() + "/workflow-instances",
            method: "POST",
            async: false,
            contentType: "application/json",
            headers: {
              "X-CSRF-Token": this._fetchToken(),
            },
            data: JSON.stringify(data),
            success: function (result, xhr, data) {
              model.setProperty(
                "/apiResponse",
                JSON.stringify(result, null, 4)
              );
            },
            error: function (request, status, error) {
              var response = JSON.parse(request.responseText);
              model.setProperty(
                "/apiResponse",
                JSON.stringify(response, null, 4)
              );
            },
          });
        },

        _fetchToken: function () {
          var fetchedToken;

          jQuery.ajax({
            url: this._getWorkflowRuntimeBaseURL() + "/xsrf-token",
            method: "GET",
            async: false,
            headers: {
              "X-CSRF-Token": "Fetch",
            },
            success(result, xhr, data) {
              fetchedToken = data.getResponseHeader("X-CSRF-Token");
            },
          });
          return fetchedToken;
        },

        _getWorkflowRuntimeBaseURL: function () {
          var appId = this.getOwnerComponent().getManifestEntry("/sap.app/id");
          var appPath = appId.replaceAll(".", "/");
          var appModulePath = jQuery.sap.getModulePath(appPath);
        //   var appModulePath = '/wf/credit_limit_apply.zccsuicreditlimitapply-0.0.1';
          return appModulePath + "/bpmworkflowruntime/v1";  
        },
      }
    );
  }
);
